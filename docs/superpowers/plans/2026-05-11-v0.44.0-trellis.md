> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `TRELLISRuntime` to the existing `3d/generation` modality so the curated TRELLIS entry actually works end-to-end. Add the TRELLIS family entry to the v0.43.x registry, fix a latent curated-entry capability lie (`supports_text_to_3d: true` on an image-only model), and annotate Wonder3D as deferred indefinitely.

**Architecture:** Plug into the family-registry pattern locked down in v0.43.3. One new `_Family` entry with `trust_remote_code=True`. One new runtime file wrapping the official TRELLIS SDK via HF AutoPipeline. The runtime verifies the real downloaded SDK API at implementation time (not the speculative shape in the plan), per the v0.43.0 Shap-E lesson.

**Tech Stack:** Python 3.10+, torch >= 2.1, transformers >= 4.46 (with trust_remote_code), diffusers >= 0.27.0, trimesh, accelerate. Possibly xformers/kaolin/other SDK-pulled extras (verified at implementation time).

```python
# src/muse/modalities/model_3d_generation/runtimes/trellis.py
"""TRELLISRuntime: image-to-3D via Microsoft's TRELLIS SDK.

Wraps the official TRELLIS pipeline loaded via HF's AutoPipeline with
trust_remote_code=True. The SDK source is published in the TRELLIS HF
repo (transformers_modules/<repo>/) and downloaded with weights on
muse pull.

API DISCOVERY: The exact pipeline class name and output shape were
verified against the downloaded TRELLIS SDK at implementation time
(NOT inferred from the design spec). The mock in the test file
reflects the real API. If TRELLIS upstream API changes, only the
sentinel names and call paths need updating; the runtime structure
stays.

Deferred-imports pattern: sentinels populated by _ensure_deps. Tests
patch sentinels directly; _ensure_deps short-circuits on non-None.
"""
from __future__ import annotations

import logging
from typing import Any

from muse.core.runtime_helpers import (
    LoadTimer, dtype_for_name, select_device, set_inference_mode,
)
from muse.modalities.model_3d_generation.protocol import Generation3DResult


logger = logging.getLogger(__name__)


torch: Any = None
_TRELLIS_PIPELINE: Any = None  # The exact loader class name verified at impl time
trimesh: Any = None
_LAST_IMPORT_ERROR: Exception | None = None


def _ensure_deps() -> None:
    """Lazy-import torch + TRELLIS pipeline + trimesh.

    The TRELLIS pipeline class is loaded via the HF AutoPipeline
    mechanism with trust_remote_code=True at runtime.from_pretrained
    time, NOT here. This function only verifies the underlying
    libraries are present so the constructor can give an actionable
    error message before reaching from_pretrained.
    """
    global torch, _TRELLIS_PIPELINE, trimesh, _LAST_IMPORT_ERROR
    if torch is None:
        try:
            import torch as _t
            torch = _t
        except Exception as e:  # noqa: BLE001
            logger.debug("TRELLISRuntime torch unavailable: %s", e)
            _LAST_IMPORT_ERROR = e
    if _TRELLIS_PIPELINE is None:
        # Try transformers AutoPipeline; the real class is registered
        # via trust_remote_code at from_pretrained time. We probe for
        # the dependency that owns AutoPipeline.
        try:
            # AutoPipeline lives in transformers' diffusion-pipelines surface;
            # the exact import path is verified at impl time.
            from transformers import AutoModel as _ap  # placeholder; verify
            _TRELLIS_PIPELINE = _ap
        except Exception as e:  # noqa: BLE001
            logger.debug("TRELLISRuntime AutoPipeline unavailable: %s", e)
            _LAST_IMPORT_ERROR = e
    if trimesh is None:
        try:
            import trimesh as _tm
            trimesh = _tm
        except Exception as e:  # noqa: BLE001
            logger.debug("TRELLISRuntime trimesh unavailable: %s", e)
            _LAST_IMPORT_ERROR = e


class TRELLISRuntime:
    """Image-to-3D runtime over TRELLIS via trust_remote_code."""

    model_id: str
    supports_image_to_3d: bool = True
    supports_text_to_3d: bool = False

    def __init__(
        self,
        *,
        model_id: str,
        hf_repo: str,
        local_dir: str | None = None,
        device: str = "auto",
        dtype: str = "fp16",
        trust_remote_code: bool = True,
        seed: int | None = None,
        sparse_structure_steps: int = 12,
        slat_steps: int = 12,
        **_: Any,
    ) -> None:
        _ensure_deps()
        if torch is None:
            raise RuntimeError(
                "torch is not installed; run "
                f"`muse models refresh {model_id}` or install "
                "`torch>=2.1.0` into this venv"
            )
        if _TRELLIS_PIPELINE is None:
            raise RuntimeError(
                "TRELLIS SDK not loadable; run "
                f"`muse models refresh {model_id}`. The SDK loads via "
                "transformers' AutoPipeline with trust_remote_code=True."
            )
        if trimesh is None:
            raise RuntimeError(
                "trimesh is not installed; needed for GLB export. "
                f"Run `muse models refresh {model_id}`."
            )
        self.model_id = model_id
        self._device = select_device(device, torch_module=torch)
        self._dtype = dtype_for_name(dtype, torch_module=torch)
        self._trust_remote_code = bool(trust_remote_code)
        self._default_seed = seed
        self._default_sparse_structure_steps = int(sparse_structure_steps)
        self._default_slat_steps = int(slat_steps)
        src = local_dir or hf_repo
        with LoadTimer(f"loading TRELLIS from {src}", logger):
            # Implementer: confirm the exact loader class against the
            # downloaded SDK at impl time. The mock pattern in
            # test_trellis.py demonstrates the expected from_pretrained
            # surface; the real call will look like:
            #   self._pipeline = _TRELLIS_PIPELINE.from_pretrained(
            #       src, trust_remote_code=trust_remote_code,
            #       torch_dtype=self._dtype,
            #   )
            self._pipeline = _TRELLIS_PIPELINE.from_pretrained(
                src,
                trust_remote_code=self._trust_remote_code,
                torch_dtype=self._dtype,
            )
            if hasattr(self._pipeline, "to"):
                self._pipeline = self._pipeline.to(self._device)
        set_inference_mode(self._pipeline)

    def image_to_3d(
        self, image: Any, **kwargs: Any,
    ) -> list[Generation3DResult]:
        raise NotImplementedError

    def text_to_3d(
        self, prompt: str, **kwargs: Any,
    ) -> list[Generation3DResult]:
        raise NotImplementedError(
            "TRELLIS-image-large is image-only. The TRELLIS text-to-3D "
            "variant has not been curated; pull a text-capable TRELLIS "
            "variant (when available) and register it separately."
        )
```

```python
"""Slow e2e: boot in-process supervisor with trellis-image enabled,
issue an image-to-3D request, assert GLB bytes returned.

Skipped when:
  - trellis-image weights not cached locally (~/.cache/huggingface/hub/

TRELLIS is GPU-required; CPU inference is impractical (multiple minutes
per request).

import base64
import os
import pathlib
import re
import subprocess
import sys
import time
from io import BytesIO

import pytest
import requests


def _trellis_weights_cached() -> bool:
    hf_cache = pathlib.Path(
        os.environ.get("HF_HOME", pathlib.Path.home() / ".cache" / "huggingface")
    )
    candidate = hf_cache / "hub" / "models--JeffreyXiang--TRELLIS-image-large"
    if not candidate.exists():
        return False
    # Verify at least one snapshot dir has a config.json or pipeline.py
    for snapshot in (candidate / "snapshots").glob("*"):
        if (snapshot / "config.json").exists() or (snapshot / "pipeline.py").exists():
            return True
    return False


def _gpu_available() -> bool:
    try:
        import torch
        return torch.cuda.is_available()
    except ImportError:
        return False


def _red_square_data_url(side: int = 16) -> str:
    """Build a tiny red-square PNG data URL for testing."""
    try:
        from PIL import Image
    except ImportError:
        pytest.skip("Pillow not available in test process")
    img = Image.new("RGB", (side, side), "red")
    buf = BytesIO()
    img.save(buf, format="PNG")
    return "data:image/png;base64," + base64.b64encode(buf.getvalue()).decode()


_TRELLIS_INSTALLED = _trellis_weights_cached()
_GPU_AVAILABLE = _gpu_available()


@pytest.mark.slow
@pytest.mark.timeout(600)
@pytest.mark.skipif(
    not _TRELLIS_INSTALLED,
    reason="trellis-image weights not cached at ~/.cache/huggingface/hub",
)
@pytest.mark.skipif(
    not _GPU_AVAILABLE,
    reason="no CUDA device available; trellis-image requires cuda",
)
def test_supervisor_serves_trellis_image_to_3d():
    # Spawn `muse serve` as a subprocess on a free port.
    port = 18792  # avoid collisions with other slow e2e tests (18790, 18791)
    proc = subprocess.Popen(
        [sys.executable, "-m", "muse.cli", "serve", "--port", str(port)],
        env={**os.environ},
        stdout=subprocess.PIPE,
        stderr=subprocess.STDOUT,
        text=True,
    )
    base_url = f"http://127.0.0.1:{port}"
    try:
        # Wait for /health.
        for _ in range(60):
            try:
                r = requests.get(f"{base_url}/health", timeout=2)
                if r.status_code == 200:
                    break
            except requests.RequestException:
                pass
            time.sleep(1)
        else:
            pytest.fail("supervisor never became ready")

        # Verify trellis-image is in /v1/models.
        r = requests.get(f"{base_url}/v1/models", timeout=10)
        r.raise_for_status()
        ids = {m["id"] for m in r.json()["data"]}
        assert "trellis-image" in ids

        # Issue image-to-3D request via the /v1/3d/from-image multipart route.
        # The image is uploaded as multipart; trellis-image's runtime
        # consumes a PIL.Image.
        from PIL import Image
        img = Image.new("RGB", (256, 256), "red")
        buf = BytesIO()
        img.save(buf, format="PNG")
        buf.seek(0)

        r = requests.post(
            f"{base_url}/v1/3d/from-image",
            files={"image": ("input.png", buf.getvalue(), "image/png")},
            data={"model": "trellis-image"},
            timeout=480,
        )
        r.raise_for_status()
        body = r.json()
        assert body["model"] == "trellis-image"
        assert len(body["data"]) >= 1

        # Decode the b64_json and verify GLB magic bytes.
        glb_b64 = body["data"][0].get("b64_json")
        assert glb_b64
        glb_bytes = base64.b64decode(glb_b64)
        assert glb_bytes[:4] == b"glTF", "response is not a valid GLB"
    finally:
        proc.terminate()
        try:
            proc.wait(timeout=10)
        except subprocess.TimeoutExpired:
            proc.kill()
```
