```python
def enable_model(model_id: str, ...) -> dict:
    # 1. fast catalog flip + duplicate guard under lock
    state = get_supervisor_state()
    with state.lock:
        set_enabled(model_id, True)
        for existing in state.workers:
            if existing.model_id == model_id:
                # Duplicate enable: surface the existing job_id (γ-flavor
                # idempotency). Spawn happens once even under concurrent
                # callers.
                if existing.job_id:
                    return {"job_id": existing.job_id, "status": "already_pending"}
                return {"status": "already_running"}
        port = find_free_port(...)
        job = job_store.create(target_kind="enable", target_id=model_id)
        spec = WorkerSpec(
            model_id=model_id, port=port, status="pending",
            job_id=job.id, ...
        )
        state.workers.append(spec)

    # 2. spawn + wait OUTSIDE the lock
    try:
        spawn_worker(spec)
        wait_for_ready(spec, timeout=120.0)
    except Exception as e:
        with state.lock:
            spec.status = "dead"
        job_store.fail(job.id, str(e))
        raise

    # 3. flip status under the lock (briefly)
    with state.lock:
        spec.status = "running"
    job_store.complete(job.id)
    return {"job_id": job.id}
```

```python
def test_concurrent_enable_coalesces_to_one_spawn(monkeypatch):
    """Two concurrent enables for the same model must NOT spawn two
    workers. The second caller observes the first caller's job_id."""
    import threading
    from muse.admin import operations

    spawned: list = []

    def _slow_spawn(spec):
        spawned.append(spec)
        # Simulate a slow GGUF load.
        import time
        time.sleep(0.3)

    def _ok_wait(spec, timeout):
        return True

    monkeypatch.setattr(operations, "spawn_worker", _slow_spawn)
    monkeypatch.setattr(operations, "wait_for_ready", _ok_wait)

    results: list = []
    def _call():
        results.append(operations.enable_model("test-model"))

    t1 = threading.Thread(target=_call)
    t2 = threading.Thread(target=_call)
    t1.start()
    # Brief delay so t1 takes the lock first.
    import time as _t
    _t.sleep(0.05)
    t2.start()
    t1.join()
    t2.join()

    assert len(spawned) == 1, f"expected 1 spawn, got {len(spawned)}"
    job_ids = {r.get("job_id") for r in results if r.get("job_id")}
    assert len(job_ids) == 1, f"expected one shared job_id, got {job_ids}"


def test_admin_lock_released_during_spawn(monkeypatch):
    """Other admin reads should not block while enable's spawn is in
    flight. Acquire-and-release the lock during the slow spawn window
    and confirm the read returns promptly."""
    import threading
    import time
    from muse.admin import operations

    def _slow_spawn(spec):
        time.sleep(0.5)

    def _ok_wait(spec, timeout):
        return True

    monkeypatch.setattr(operations, "spawn_worker", _slow_spawn)
    monkeypatch.setattr(operations, "wait_for_ready", _ok_wait)

    state = operations.get_supervisor_state()

    enable_done = threading.Event()
    def _enable():
        operations.enable_model("test-model-2")
        enable_done.set()

    threading.Thread(target=_enable).start()
    time.sleep(0.1)  # give enable a head start

    # Snapshot workers should return promptly even though enable is
    # mid-spawn. We measure how long this acquire takes.
    t0 = time.perf_counter()
    with state.lock:
        snapshot = list(state.workers)
    elapsed = time.perf_counter() - t0
    assert elapsed < 0.05, (
        f"state.lock was held during spawn for {elapsed:.3f}s; "
        "should release between append-pending and spawn"
    )

    enable_done.wait(timeout=2.0)
```
