```python
@dataclass
class ClassificationResult:
    """One model's classification of one or more inputs.

    `results` is always a list of lists (one inner list per input),
    even when the original request had a single string input. Each
    inner list is sorted by score descending. This shape is uniform
    across fine-tuned classifiers and zero-shot, so the codec doesn't
    branch on runtime type.
    """
    model_id: str
    results: list[list[dict]]  # [[{label: str, score: float}, ...], ...]


class ZeroShotClassifier(Protocol):
    """Optional protocol for runtimes that support zero-shot dispatch.

    The route checks `supports_zero_shot` capability before calling
    classify_zero_shot; runtimes don't need to advertise this Protocol
    type explicitly. The wire shape is identical to classify (returns
    a per-input list of {label, score} pairs).
    """

    def classify_zero_shot(
        self,
        inputs: list[str],
        candidate_labels: list[str],
        *,
        multi_label: bool = False,
        top_k: int | None = None,
    ) -> ClassificationResult: ...


class TextClassifier(Protocol):
    """For fine-tuned classifiers; sentiment, intent, etc.

    The existing /v1/moderations route uses this same Protocol; the
    new /v1/text/classifications route reads the full label distribution
    rather than reducing to flag/score.
    """

    def classify(
        self,
        inputs: list[str],
        *,
        top_k: int | None = None,
    ) -> ClassificationResult: ...
```

```python
def truncate_top_k(
    per_input_labels: list[list[dict]], k: int | None,
) -> list[list[dict]]:
    """Sort each inner list by score desc; if k is set, keep only top-k.

    Stable: if the runtime returns labels in a deterministic order,
    truncate_top_k preserves that order among ties (Python's sort
    is stable).
    """
    out: list[list[dict]] = []
    for labels in per_input_labels:
        sorted_labels = sorted(labels, key=lambda d: d["score"], reverse=True)
        if k is not None and k > 0:
            sorted_labels = sorted_labels[:k]
        out.append(sorted_labels)
    return out


def classification_to_openai_dict(result: ClassificationResult) -> dict:
    """Build the wire envelope for /v1/text/classifications."""
    import time as _t
    return {
        "id": f"classify-{int(_t.time() * 1000)}",
        "model": result.model_id,
        "results": result.results,
    }
```

```python
"""Generic runtime for zero-shot text classification.

Wraps transformers.pipeline("zero-shot-classification") which handles
the NLI premise/hypothesis construction internally. Approximately 80
LOC because the pipeline does the fiddly work.

Deferred-imports discipline: torch and transformers are module-level
sentinels. _ensure_deps() lazy-imports inside the constructor. Tests
patch the module-level names directly.
"""
from __future__ import annotations

import logging
from typing import Any

from muse.core.runtime_helpers import (
    LoadTimer, dtype_for_name, select_device, set_inference_mode,
)
from muse.modalities.text_classification.protocol import ClassificationResult


logger = logging.getLogger(__name__)


# Module-level sentinels. _ensure_deps populates them; tests pre-populate
# to bypass the real imports.
torch = None
pipeline = None


def _ensure_deps() -> None:
    global torch, pipeline
    if torch is None:
        try:
            import torch as _torch
            torch = _torch
        except ImportError as e:  # noqa: BLE001
            logger.debug("torch import failed: %s", e)
    if pipeline is None:
        try:
            from transformers import pipeline as _pipeline
            pipeline = _pipeline
        except ImportError as e:  # noqa: BLE001
            logger.debug("transformers.pipeline import failed: %s", e)


class HFZeroShotPipeline:
    """Wraps `pipeline("zero-shot-classification")` for any NLI head."""

    def __init__(
        self,
        *,
        hf_repo: str | None = None,
        local_dir: str | None = None,
        device: str = "auto",
        dtype: str = "fp32",
        model_id: str | None = None,
        **_: Any,
    ) -> None:
        _ensure_deps()
        if torch is None:
            raise RuntimeError(
                "torch is not installed; run `muse pull <model>` to install"
            )
        if pipeline is None:
            raise RuntimeError(
                "transformers is not installed; run `muse pull <model>` "
                "to install"
            )
        self.model_id = model_id or "zero-shot"
        self._device = self._select_device(device)
        self._dtype = self._resolve_dtype(dtype)
        self._source = local_dir or hf_repo
        if self._source is None:
            raise ValueError(
                "HFZeroShotPipeline requires hf_repo or local_dir"
            )
        with LoadTimer(logger, f"loading zero-shot pipeline from {self._source}"):
            self._pipe = pipeline(
                task="zero-shot-classification",
                model=self._source,
                device=self._device,
                torch_dtype=self._dtype,
            )
        set_inference_mode(self._pipe.model, torch_module=torch)

    # Thin delegators for test-patch compatibility (matches sibling
    # runtimes; flagged by the meta-test if open-coded).
    def _select_device(self, device: str) -> str:
        return select_device(device, torch_module=torch)

    def _resolve_dtype(self, dtype: str):
        return dtype_for_name(dtype, torch_module=torch)

    def classify_zero_shot(
        self,
        inputs: list[str],
        candidate_labels: list[str],
        *,
        multi_label: bool = False,
        top_k: int | None = None,
    ) -> ClassificationResult:
        """Run zero-shot NLI on each input.

        The pipeline returns one dict per input with `labels` (sorted
        by score desc) and `scores`. We rebuild as the standard
        per-input list of {label, score}, applying top_k truncation
        in the codec layer.
        """
        if not candidate_labels:
            raise ValueError("candidate_labels must be non-empty")
        out_rows: list[list[dict]] = []
        # The pipeline accepts a list of inputs natively but we iterate
        # explicitly so a malformed input doesn't poison the batch.
        for text in inputs:
            raw = self._pipe(
                text,
                candidate_labels=candidate_labels,
                multi_label=multi_label,
            )
            labels = raw["labels"]
            scores = raw["scores"]
            row = [
                {"label": l, "score": float(s)}
                for l, s in zip(labels, scores)
            ]
            if top_k is not None and top_k > 0:
                row = row[:top_k]
            out_rows.append(row)
        return ClassificationResult(model_id=self.model_id, results=out_rows)
```

```python
class _ClassificationsRequest(BaseModel):
    """Request shape for /v1/text/classifications.

    `input` is a string or list of strings. `candidate_labels`
    presence dispatches to zero-shot. `top_k` truncates the per-input
    label list. `multi_label` is forwarded to zero-shot only.
    """
    input: str | list[str]
    model: str | None = None
    candidate_labels: list[str] | None = None
    top_k: int | None = Field(default=None, ge=1, le=1000)
    multi_label: bool = False

    @field_validator("input")
    @classmethod
    def _input_nonempty(cls, v):
        if isinstance(v, str):
            if not v:
                raise ValueError("input string cannot be empty")
        elif isinstance(v, list):
            if not v:
                raise ValueError("input list cannot be empty")
            if any(not isinstance(s, str) or not s for s in v):
                raise ValueError("input list must contain non-empty strings")
        return v

    @field_validator("candidate_labels")
    @classmethod
    def _candidate_labels_nonempty(cls, v):
        if v is not None:
            if not v:
                raise ValueError("candidate_labels list cannot be empty")
            if any(not isinstance(s, str) or not s for s in v):
                raise ValueError("candidate_labels must be non-empty strings")
        return v
```

```python
@router.post("/v1/text/classifications")
async def text_classifications(req: _ClassificationsRequest, registry_dep=Depends(...)):
    try:
        backend = registry.get(MODALITY, req.model)
    except KeyError:
        raise ModelNotFoundError(model_id=req.model or "<default>", modality=MODALITY)

    manifest = registry.manifest(MODALITY, getattr(backend, "model_id", req.model)) or {}
    capabilities = manifest.get("capabilities", {})

    inputs = [req.input] if isinstance(req.input, str) else req.input

    if req.candidate_labels is not None:
        # Zero-shot dispatch
        if not capabilities.get("supports_zero_shot"):
            return error_response(
                400, "zero_shot_not_supported",
                f"model {getattr(backend, 'model_id', '?')} does not support "
                f"zero-shot (no supports_zero_shot capability flag); "
                f"omit candidate_labels or use a zero-shot model",
            )
        def _call():
            with backend._inference_lock:
                return backend.classify_zero_shot(
                    inputs,
                    candidate_labels=req.candidate_labels,
                    multi_label=req.multi_label,
                    top_k=req.top_k,
                )
    else:
        if not capabilities.get("supports_classification"):
            return error_response(
                400, "candidate_labels_required",
                f"model {getattr(backend, 'model_id', '?')} requires "
                f"candidate_labels (zero-shot only); pass candidate_labels "
                f"or use a fine-tuned classifier",
            )
        def _call():
            with backend._inference_lock:
                return backend.classify(inputs, top_k=req.top_k)

    try:
        result = await asyncio.to_thread(_call)
    except Exception as e:  # noqa: BLE001
        logger.exception("classify failed")
        return error_response(500, "internal_error", str(e))

    return classification_to_openai_dict(result)
```

```python
class ClassificationsClient:
    """HTTP client for /v1/text/classifications.

    Mirrors the route's wire shape. base_url defaults from MUSE_SERVER
    env var like sibling clients (EmbeddingsClient, ModerationsClient).
    """

    def __init__(self, base_url: str | None = None):
        self.base_url = (
            base_url or os.environ.get("MUSE_SERVER", "http://localhost:8000")
        )

    def classify(
        self,
        input: str | list[str],
        *,
        model: str | None = None,
        candidate_labels: list[str] | None = None,
        top_k: int | None = None,
        multi_label: bool = False,
    ) -> dict:
        body = {"input": input, "multi_label": multi_label}
        if model:
            body["model"] = model
        if candidate_labels is not None:
            body["candidate_labels"] = candidate_labels
        if top_k is not None:
            body["top_k"] = top_k
        with httpx.Client(timeout=300.0) as client:
            r = client.post(f"{self.base_url}/v1/text/classifications", json=body)
            r.raise_for_status()
            return r.json()
```

```python
def _is_zero_shot(info) -> bool:
    tags = getattr(info, "tags", None) or []
    if "zero-shot-classification" in tags:
        return True
    # Repo name fallback for repos that ship under text-classification
    # tag only (some NLI checkpoints).
    repo_id = (getattr(info, "id", "") or "").lower()
    return any(s in repo_id for s in ("zero-shot", "mnli", "nli"))


def _resolve(repo_id: str, variant: str | None, info) -> ResolvedModel:
    is_zs = _is_zero_shot(info)
    runtime_path = (
        "muse.modalities.text_classification.runtimes.hf_zero_shot:HFZeroShotPipeline"
        if is_zs else
        "muse.modalities.text_classification.runtimes.hf_text_classifier:HFTextClassifier"
    )
    capabilities = {
        "device": "auto",
        "supports_zero_shot": is_zs,
        "supports_classification": not is_zs,
    }
    manifest = {
        "model_id": _model_id(repo_id),
        "modality": "text/classification",
        "hf_repo": repo_id,
        "description": (
            f"Zero-shot classifier: {repo_id}" if is_zs
            else f"Text classifier: {repo_id}"
        ),
        ...
        "capabilities": capabilities,
    }
    ...
```
