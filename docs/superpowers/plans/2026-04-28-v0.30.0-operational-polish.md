### Changes to `supervisor.py`

Add helper `_wait_for_first_ready`:

```python
def _wait_for_first_ready(
    specs: list[WorkerSpec],
    *,
    timeout: float = 60.0,
    poll_interval: float = 0.5,
) -> WorkerSpec:
    """Block until ANY worker passes /health.

    Returns the first WorkerSpec that becomes ready. Round-robin
    polling, so a fast worker buried behind a slower spec still wins.

    Raises TimeoutError if none pass within timeout. The caller
    should treat that as a fatal supervisor-bringup failure: every
    spawned worker is sick, gateway should not start.
    """
    if not specs:
        raise ValueError("_wait_for_first_ready called with no specs")
    deadline = time.monotonic() + timeout
    last_err: Exception | None = None
    while time.monotonic() < deadline:
        for spec in specs:
            try:
                r = httpx.get(f"http://127.0.0.1:{spec.port}/health", timeout=2.0)
                if r.status_code == 200:
                    return spec
            except httpx.HTTPError as e:
                last_err = e
        time.sleep(poll_interval)
    raise TimeoutError(
        f"no worker became ready within {timeout}s "
        f"(spawned {len(specs)}; last error: {last_err})"
    )
```

Add helper `_promote_workers`:

```python
def _promote_workers(
    specs: list[WorkerSpec],
    state: SupervisorState,
    *,
    timeout: float = 120.0,
) -> None:
    """Background-thread target: poll /health for late-loading workers.

    Promotes spec.status from 'pending' to 'running' once /health
    responds 200. On timeout, sets status='unhealthy'; the
    auto-restart monitor's next tick will trigger a respawn.

    Acquires state.lock for status mutations so admin readers and
    /v1/models aggregations see consistent values.
    """
    for spec in specs:
        try:
            wait_for_ready(port=spec.port, timeout=timeout)
            with state.lock:
                spec.status = "running"
            logger.info("late-promote: worker on port %d ready", spec.port)
        except TimeoutError:
            with state.lock:
                spec.status = "unhealthy"
            logger.warning(
                "late-promote: worker on port %d did not become ready in %ds; "
                "auto-restart monitor will retry",
                spec.port, timeout,
            )
```

Refactor `run_supervisor` from "spawn all + wait all + start gateway"
to "spawn all + wait first + start gateway + promote rest in
background":

```python
def run_supervisor(*, host: str, port: int, device: str) -> int:
    from muse.cli_impl.gateway import build_gateway

    specs = plan_workers()
    if not specs:
        logger.warning(
            "no pulled models with a venv - server will start empty. "
            "Pull a model first: `muse pull <model-id>`"
        )

    state = SupervisorState(workers=specs, device=device)
    set_supervisor_state(state)

    stop_event = threading.Event()
    monitor_thread: threading.Thread | None = None
    boot_thread: threading.Thread | None = None

    try:
        for spec in specs:
            spawn_worker(spec, device=device)

        # Wait for the FIRST worker only. With one spec the loop
        # terminates on iteration one; with N it returns the moment
        # any single worker passes /health. The remaining workers
        # promote on a background thread (boot_thread) so the
        # gateway can serve fast workers while slow ones load.
        if specs:
            first_ready = _wait_for_first_ready(specs)
            with state.lock:
                first_ready.status = "running"
            logger.info(
                "first worker ready on port %d (%s); gateway will start now; "
                "remaining %d worker(s) will promote in the background",
                first_ready.port, first_ready.models, len(specs) - 1,
            )

            remaining = [s for s in specs if s is not first_ready]
            if remaining:
                boot_thread = threading.Thread(
                    target=_promote_workers,
                    args=(remaining, state),
                    daemon=True,
                    name="muse-late-boot",
                )
                boot_thread.start()

            monitor_thread = threading.Thread(
                target=_monitor_workers,
                args=(state.workers, stop_event),
                daemon=True,
                name="muse-monitor",
            )
            monitor_thread.start()
            logger.info(
                "auto-restart monitor running (interval=%.1fs, threshold=%d, budget=%d)",
                _MONITOR_INTERVAL, _FAILURE_THRESHOLD, _MAX_RESTARTS,
            )

        # Build gateway with a live reference to state. Routes are
        # derived per-request from state.workers (running-only) so
        # late-promoting workers join the routing table without a
        # reload.
        app = build_gateway(state=state)

        logger.info("starting gateway on %s:%d", host, port)
        uvicorn.run(app, host=host, port=port, log_config=None)
    except KeyboardInterrupt:
        logger.info("shutting down (SIGINT)")
    finally:
        stop_event.set()
        if monitor_thread is not None:
            monitor_thread.join(timeout=5.0)
        # boot_thread is daemon=True so it dies with the process; no
        # join needed (it's polling httpx with short timeouts and
        # will exit on its own).
        _shutdown_workers(state.workers)
        clear_supervisor_state()
        try:
            from muse.admin.jobs import get_default_store
            get_default_store().shutdown()
        except Exception as e:  # noqa: BLE001
            logger.warning("admin job-store shutdown failed: %s", e)
    return 0
```

### Changes to `gateway.py`

`build_gateway` previously took `routes: list[WorkerRoute]`. We add a
`state: SupervisorState | None` parameter that, when supplied, is
used as the live source of truth for routing. When omitted (legacy
call path used by tests), the old `routes` list still works.

```python
def build_gateway(
    routes: list[WorkerRoute] | None = None,
    *,
    state: "SupervisorState | None" = None,
    timeout: float = 300.0,
) -> FastAPI:
    """Build the gateway FastAPI app.

    Two routing modes:
      - state-driven (preferred): pass `state` (a SupervisorState).
        Routes are re-derived per request from state.workers, so
        late-promoting workers join the routing table without an
        app rebuild.
      - static: pass `routes` (a fixed list of WorkerRoute). Used by
        legacy tests that want a frozen routing table.
    """
    ...
    app.state.routes_state = state
    app.state.static_routes = (
        {r.model_id: r for r in routes} if routes is not None else None
    )

    def _routes_now() -> dict[str, WorkerRoute]:
        if app.state.routes_state is not None:
            out: dict[str, WorkerRoute] = {}
            with app.state.routes_state.lock:
                for spec in app.state.routes_state.workers:
                    if spec.status != "running":
                        continue
                    url = f"http://127.0.0.1:{spec.port}"
                    for m in spec.models:
                        out[m] = WorkerRoute(model_id=m, worker_url=url)
            return out
        return app.state.static_routes or {}

    app.state.routes_now = _routes_now
    ...
```

In each route handler that previously read `app.state.routes`:


### Regression-guard test

`tests/models/test_pip_extras_audit.py`:

```python
"""Static audit: bundled-script pip_extras tuples cover their runtime imports.

Reads each src/muse/models/*.py via ast.parse, walks the import nodes
(top-level + inside-function via _ImportWalker), and maps each module
to its PyPI dist via MODULE_TO_PYPI. Asserts every PyPI dist is
declared in the script's MANIFEST['pip_extras'].

This is a "best effort" check: runtimes that load modules via
importlib are out of scope (none in muse today). The audit defends
against the most common pip_extras gap: a runtime module is added,
its top-level `from X import Y` is added, but pip_extras isn't
updated.
"""
from __future__ import annotations

import ast
import importlib
import re
from pathlib import Path

import pytest


# Maps a Python top-level module name to the PyPI distribution name
# that provides it. Build incrementally; new bundled scripts add
# entries.
MODULE_TO_PYPI: dict[str, str] = {
    "torch": "torch",
    "transformers": "transformers",
    "diffusers": "diffusers",
    "accelerate": "accelerate",
    "PIL": "Pillow",
    "safetensors": "safetensors",
    "sentence_transformers": "sentence-transformers",
    "scipy": "scipy",
    "inflect": "inflect",
    "unidecode": "unidecode",
    "numpy": "numpy",
    "librosa": "librosa",
    "kokoro": "kokoro",
    "misaki": "misaki",
    "soundfile": "soundfile",
    "einops": "einops",
    "imageio": "imageio",
}

# Modules to skip entirely (stdlib, this project, internal deps).
SKIP_PREFIXES = ("muse", "_", "logging", "io", "math", "os", "sys",
                 "pathlib", "typing", "dataclasses", "json",
                 "subprocess", "shutil", "warnings", "threading",
                 "time", "tempfile", "functools", "itertools",
                 "collections", "contextlib", "re", "string",
                 "abc", "enum", "uuid", "datetime")
STDLIB_NAMES = {"__future__"}


def _bundled_scripts() -> list[Path]:
    here = Path(__file__).resolve().parents[2]
    models_dir = here / "src" / "muse" / "models"
    return sorted(p for p in models_dir.glob("*.py") if p.name != "__init__.py")


def _walk_imports(tree: ast.AST) -> set[str]:
    """Return the set of top-level module names imported in tree.

    Handles:
      - import X.Y         -> X
      - import X.Y as Z    -> X
      - from X.Y import Z  -> X
    """
    out: set[str] = set()
    for node in ast.walk(tree):
        if isinstance(node, ast.Import):
            for alias in node.names:
                out.add(alias.name.split(".", 1)[0])
        elif isinstance(node, ast.ImportFrom):
            if node.module and node.level == 0:
                out.add(node.module.split(".", 1)[0])
    return out


def _read_manifest(script: Path) -> dict:
    """Parse MANIFEST = { ... } from a script via importlib.

    The bundled scripts declare MANIFEST at module-top with no heavy
    imports gated by it; importing the module is safe.
    """
    rel = "muse.models." + script.stem
    mod = importlib.import_module(rel)
    return getattr(mod, "MANIFEST", {})


def _strip_pip_extras_to_dists(extras: tuple[str, ...]) -> set[str]:
    """('torch>=2.1', 'sentence-transformers>=2', 'misaki[en]') ->
       {'torch', 'sentence-transformers', 'misaki'}.

    Drops version specifiers and bracketed extras.
    """
    out: set[str] = set()
    for entry in extras:
        # Strip [extras] (e.g. 'misaki[en]', 'imageio[ffmpeg]')
        bare = re.sub(r"\[.*?\]", "", entry)
        # Strip version specifier (e.g. '>=2.1.0', '==4.36.0')
        bare = re.split(r"[<>=!~]", bare, maxsplit=1)[0]
        out.add(bare.strip())
    return out


def _classify_module(name: str) -> str | None:
    """Return PyPI dist name for `name`, or None if module should be skipped.

    Raises pytest.fail if the module isn't classified anywhere; this
    forces the maintainer to update MODULE_TO_PYPI when a new bundled
    script imports a new module.
    """
    if name in MODULE_TO_PYPI:
        return MODULE_TO_PYPI[name]
    if name in STDLIB_NAMES:
        return None
    if any(name == p or name.startswith(p + ".") for p in SKIP_PREFIXES):
        return None
    if name.startswith("_"):
        return None
    pytest.fail(
        f"unknown module {name!r}: add to MODULE_TO_PYPI in "
        f"tests/models/test_pip_extras_audit.py to enable the audit, "
        f"or to SKIP_PREFIXES if it's stdlib/local"
    )


@pytest.mark.parametrize("script", _bundled_scripts(), ids=lambda p: p.stem)
def test_pip_extras_covers_runtime_imports(script: Path):
    """Every bundled script's pip_extras includes the PyPI dists for
    its runtime imports."""
    src = script.read_text()
    tree = ast.parse(src)
    imported = _walk_imports(tree)
    needed: set[str] = set()
    for module_name in imported:
        dist = _classify_module(module_name)
        if dist:
            needed.add(dist)

    manifest = _read_manifest(script)
    declared = _strip_pip_extras_to_dists(tuple(manifest.get("pip_extras", ())))

    # Synonyms: PIL imports map to Pillow, etc. Already handled by
    # MODULE_TO_PYPI. Compare bare PyPI dist names.
    missing = needed - declared
    assert not missing, (
        f"{script.name}: pip_extras is missing {sorted(missing)}; "
        f"runtime imports {sorted(needed)} but declared {sorted(declared)}"
    )
```
